    # 拓展nuxt的路由，将多个地址指向同一个vue生成的动态路由

    # 官方教程代码：




     









    const resolve = require('path').resolve
module.exports = {
  router: {
    extendRoutes (routes) {
      routes.push({
        name: 'custom',
        path: '*',
        component: resolve(__dirname, 'pages/404.vue')
      })
    }
  }
}
